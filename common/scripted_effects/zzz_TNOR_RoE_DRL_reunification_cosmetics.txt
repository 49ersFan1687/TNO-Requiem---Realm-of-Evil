DRL_update_reunification_cosmetic_tag = {

	hidden_effect = {
		drop_cosmetic_tag = yes
	}
	
	if = {
		limit = {
			original_tag = DRL
			has_global_flag = RUS_national_unification
			NOT = { has_country_flag = RUS_regional_stage }
		}
		set_cosmetic_tag = DIR_RUSSIA
	}
	else_if = {
		limit = {
			original_tag = DRL
			has_country_flag = RUS_superregional_stage
			NOT = { has_country_flag = RUS_regional_stage }
			NOT = { has_global_flag = RUS_national_unification }
		}
		set_cosmetic_tag = DIR_RUSSIA
	}
	else_if = {
		limit = {
			original_tag = DRL
			has_country_flag = RUS_regional_stage
			NOT = { has_country_flag = RUS_superregional_stage }
			NOT = { has_global_flag = RUS_national_unification }
		}
		set_cosmetic_tag = DRL_Westrussia
	}
	else_if = {
		limit = {
			original_tag = DRL
			has_country_flag = DRL_southern_urals_unified
			NOT = { has_country_flag = RUS_regional_stage }
			NOT = { has_country_flag = RUS_superregional_stage }
			NOT = { has_global_flag = RUS_national_unification }
			NOT = { has_global_flag = RUS_West_Russian_unification }
		}
		set_cosmetic_tag = DIR_BEST
	}
}
